\section*{Packages}
\begin{itemize}
    \item \textbf{Assimp:} Parses imported .obj files into mesh data.
    \item \textbf{OpenTK:} C\# wrapper for OpenGL, used for rendering graphics and viewport updates.
    \item \textbf{Blender (Python API):} Used for mesh simplification and unfolding/export to .svg.
    \item \textbf{Docker:} Containerization of frontend, backend, database, and NGINX server for deployment.
\end{itemize}

\section*{Pages and Screens}
\begin{itemize}
    \item \textbf{Splash Screen:} Displays the application logo and loading status during startup.
    \item \textbf{Viewport:}
    \begin{itemize}
        \item Import button for uploading 3D models in .obj format.
        \item Render region powered by OpenTK for displaying 3D models.
        \item Simplify button with bounding-box and decimation options.
        \item Export button:
        \begin{itemize}
            \item Passes simplified model to Blender for unfolding.
            \item Produces .svg file with cuts and folds marked.
            \item Downloads .svg file to user.
            \item Displays unfolded model preview in a new window.
            \item Option to print directly from the preview window.
            \item Option to save .svg file for later use.
        \end{itemize}
        \item UI polish: error handling, loading indicators, responsive design, and edge case support.
    \end{itemize}
\end{itemize}

\section*{Workflow Summary}
\begin{enumerate}
    \item Application Startup $\rightarrow$ Splash Screen $\rightarrow$ Viewport
    \item User imports .obj file $\rightarrow$ File parsed by Assimp $\rightarrow$ Rendered in OpenTK viewport
    \item User simplifies model $\rightarrow$ Simplification options applied $\rightarrow$ Model updated in viewport
    \item User exports model $\rightarrow$ Model sent to Blender via Python script $\rightarrow$ Model unfolded and exported as .svg
    \item .svg file downloaded to user $\rightarrow$ Unfolded model preview displayed $\rightarrow$ User can print or save .svg file 
\end{enumerate}

\section*{Docker Compose Services}
\begin{itemize}
    \item \textbf{frontend:} React application served via Node.js and NGINX.
    \item \textbf{backend:} Express API for modeling logic, database interactions, and Blender Python scripting.
    \item \textbf{database:} PostgreSQL container with persistent volume.
    \item \textbf{nginx:} Reverse proxy and static file server for frontend and backend services.
\end{itemize}
